CREATE TABLE fundamentals (
		ID INTEGER PRIMARY KEY,
		symbol TEXT,
		year_ending DATE,
		cash_equival NUMERIC,
		earnings_before NUMERIC,
		gross_margin FLOAT,
		net_income NUMERIC,
		total_assets NUMERIC,
		total_liabilities NUMERIC,
		total_revenue NUMERIC,
		year INTEGER,
		earnings_per_share FLOAT,
		share_outstandidng NUMERIC

		);
		
CREATE TABLE prices (
		date DATE, 
		symbol TEXT,
		open FLOAT,
		close FLOAT,
		low FLOAT,
		high FLOAT,
		volume INTEGER
		);
		
CREATE TABLE securities (
		Symbol TEXT,
		Company TEXT,
		Sector TEXT,
		sub_industry TEXT,
		initial_date DATE
		);
